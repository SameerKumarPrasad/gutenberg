\question If $\tan\alpha = \dfrac{5}{6}$ and $\tan\beta = \dfrac{1}{11}$, then what is $(\alpha + \beta)$?

\insertQR{}

\ifprintanswers
\fi 

\begin{solution}
   \begin{align}
      \tan(\alpha + \beta) &= \dfrac{\tan\alpha + \tan\beta}{1-\tan\alpha\cdot\tan\beta} \\
          &= \dfrac{\frac{5}{6} + \frac{1}{11}}{1-\frac{5}{6}\cdot\frac{1}{11}} = 1 \\
          \Rightarrow \alpha + \beta &= \tan^{-1} 1 = \dfrac{\pi}{4}
   \end{align}
\end{solution}
