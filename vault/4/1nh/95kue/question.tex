\begin{parts}
  \part[3] Determine whether the equation has real roots

  \insertQR{}
\begin{solution}[\mcq]
  For the equation $6x^2-7x+2$ to have real roots discriminant (D) $\geq$ 0 \\
  \begin{align}
  D &= b^2 - 4\cdot a \cdot c = 1 \geq 0
  \end{align}
 Thus, the equation has real roots  
  \end{solution}

  \part[4] Find the roots by first expressing the equation in the form $\left( x - a\right)^2 - b^2 = 0$

  \insertQR{}
\begin{solution}[\halfpage]
  \begin{align}
  6x^2-7x+2=0 \rightarrow 36x^2 - 42x +12 \\
  \Rightarrow \underbrace{\left[ (6x)^2 - 2\cdot\dfrac{7}{2} \cdot 6x  + \left(\dfrac{7}{2}\right)^2 \right]}_{(a-b)^2} + \left(12 - \dfrac{49}{4}\right) \\
  \Rightarrow (6x - \dfrac{7}{2})^2 - (\dfrac{1}{2})^2 &= 0\\
  \Rightarrow x &= \dfrac{2}{3} , \dfrac{1}{2} 
  \end{align}
  \end{solution}

 
\end{parts}
