\begin{parts}
  \part[3] "$\vbthree$" is included in each word

\begin{solution}[\mcq]
  When one letter is fixed as $\vbthree$. We choose $\vbfour$ letters from the remaining letters of $\vbtwo$ other than "$\vbthree$".\\
  And we arrange these letters in $\vbone !$ ways.\\
  Total number of words when "$\vbthree$" is included in each word 
  \begin{align}
  \Rightarrow ^{7}C_{\vbone} \times {\vbone}!  = \vbsix
  \end{align}
  \end{solution}

  \part[3] "$\vbthree$" is not included in each word

\begin{solution}[\mcq]
  We choose $\vbone$ letters from the remaining letters of $\vbtwo$ other than "$\vbthree$".\\
  And we arrange these letters in $\vbone !$ ways.\\
  Total number of words when "$\vbthree$" is not included in any word 
  \begin{align}
  \Rightarrow ^{\vbfive}C_{\vbone} \times {\vbone}! = \vbseven
  \end{align}
  \end{solution}


\end{parts}
