\begin{solution}[\halfpage]
Put x = $t^2$\\
\begin{align}
&\Rightarrow dx = (2t) dt  \\
&\Rightarrow \dfrac{(2t) dt}{2t(t^2+1)}\\
&\Rightarrow \dfrac{dt}{t^2+1}\\
&\Rightarrow \tan ^ {-1} t + C 
\end{align}
Now replacing t,\\
\begin{align}
\tan ^ {-1} \sqrt{x} +C
\end{align}
\end{solution}
