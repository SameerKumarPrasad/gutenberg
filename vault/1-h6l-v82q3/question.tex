\question For what values of $x$ would $2x$, $x-8$ and $2x+5$ be successive terms 
of a geometric progression?

\calculator{1444 = 38^2}

\insertQR{}

\ifprintanswers
\fi 

\begin{solution}
	If $2x$, $x-8$ and $2x+5$ are to be successive terms of a geometric progression, then
	\begin{align}
		(x - 8)^2 &= (2x+5)\cdot 2x \\
		\Rightarrow x^2-16x + 64 &= 4x^2 + 10x \text{ or } 3x^2+26x-64 = 0 \\
		\Rightarrow x &= \dfrac{-26 \pm \sqrt{26^2-4\cdot 3\cdot(-64)}}{2\times 3} \\
		&=  \dfrac{-26 \pm 38}{6} = 2, -\frac{64}{6} = -\frac{32}{3}
	\end{align}
	
	Answer: $x=2$ or $x=-\frac{32}{3}$
\end{solution}
