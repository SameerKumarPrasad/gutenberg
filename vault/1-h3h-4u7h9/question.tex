\question Evaluate the integral $\int\dfrac{x^2+5}{x^2+2}\ud x$

\insertQR{}

\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
  \marginnote[2cm]{Using the standard result $\int\dfrac{\ud x}{x^2+a^2} = 
  \dfrac{1}{a}\tan^{-1}\dfrac{x}{a} + C$}
\fi 

\begin{solution}
	\begin{align}
		\int\dfrac{x^2+5}{x^2+2}\ud x &= \int\dfrac{(x^2+2)+3}{x^2+2}\ud x \\
		&= \int\ud x + 3\int\dfrac{\ud x}{x^2+2} \\
		&= x + \dfrac{3}{\sqrt{2}}\tan^{-1}\dfrac{x}{\sqrt{2}} + C
	\end{align}
\end{solution}
