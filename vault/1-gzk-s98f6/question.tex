\question[2]  In the adjoining figure, what is the value of $\angle ABO$ if
\insertQR{QRC}
$\angle BOC = \ang{84}$ and $\angle ACO = \ang{29}$?

\begin{solution}[\halfpage]
	In the figure, 
	\begin{align}
 		\angle BAC &= \dfrac{1}{2}\angle BOC = \dfrac{1}{2}\cdot\ang{84} = \ang{42}
	\end{align}
	
	Also, 
	\begin{align}
		\angle BAC &= \angle BAO + \angle OAC \\
		           &= \angle BAO + \ang{29} \\
		\Rightarrow \angle BAO &= \angle OBA = \ang{42} - \ang{29} = \ang{13}
	\end{align}
\end{solution}
