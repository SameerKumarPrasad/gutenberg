\question Find the sum of the infinite series $\frac{1}{5}+\frac{1}{7}+\frac{1}{5^2} + \frac{1}{7^2}
+ \frac{1}{5^3} + \frac{1}{7^3} \ldots$

\insertQR{}

\ifprintanswers
\fi 

\begin{solution}
	\begin{align}
		S &= \frac{1}{5}+\frac{1}{7}+\frac{1}{5^2} + \frac{1}{7^2} + \frac{1}{5^3} + \frac{1}{7^3} \ldots \\
		  &= \underbrace{\frac{1}{5} + \frac{1}{5^2} + \frac{1}{5^3}\ldots}_{\text{one geometric progression}}
		  + \underbrace{\frac{1}{7} + \frac{1}{7^2} + \frac{1}{7^3}\ldots}_{\text{and another}} \\
		  &= \dfrac{\frac{1}{5}}{1-\frac{1}{5}} + \dfrac{\frac{1}{7}}{1-\frac{1}{7}} \\
		  &= \dfrac{1}{4} + \dfrac{1}{6} = \dfrac{5}{12}
	\end{align}
\end{solution}
