\question[2]  In the adjoining figure, $AD \perp BC$. Prove that $AB^2+CD^2=BD^2+AC^2$
\insertQR{QRC}

\begin{marginfigure}
	\figinit{cm}
	\figpt 1: (0,0)
	\figpt 2: (4,0)
	\figpt 3: (5.5,3)
	\figptorthoprojline 12:$D$= 2/1,3/
	\figdrawbegin{}
		\figdrawline [1,2,3,1]

	\figdrawend
	\figvisu{\figBoxA}{Example}{}
	\figwritenw 1:$B$(0.1)
	\figwriten 2:$A$(0.1)
	\figwriten 3:$C$(0.2)
	\figwriten 12:$D$(0.8)
	\centerline{\box\figBoxA}
\end{marginfigure}

\begin{solution}[\halfpage]
	$\bigtriangleup ABD$ and $\bigtriangleup ACD$ are both right-angled triangles
	
	And therefore, the following hold for them	
	\begin{align}
		BD^2 + AD^2 &= AB^2 \Rightarrow AD^2 = AB^2-BD^2 \\
		CD^2 + AD^2 &= AC^2 \Rightarrow AD^2 = AC^2 - CD^2 \\
		\Rightarrow AB^2-BD^2 &= AC^2-CD^2 \\
		\Rightarrow AB^2+CD^2 &= AC^2 + BD^2
	\end{align}
	
\end{solution}
