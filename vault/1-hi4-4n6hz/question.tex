\question If $S=\overbrace{(\vbone + \vbtwo)^6}^{A} + \overbrace{(\vbone - \vbtwo)^6}^{B}$, then

\insertQR{}

\watchout

\ifprintanswers
\fi 

\begin{parts}
  \part[1] Write the binomial expansion of $A$ and then reduce it to get the value of $A$

  \insertQR{}
\begin{solution}[\mcq]
  	\begin{align}
  		A &= \sum_{k=0}^{6}\binom{6}{k}(\vbone)^{6-k}\cdot(\vbtwo)^{k} \\
  		&= \binom{6}{0}(\vbone)^{6} + \binom{6}{1}(\vbone)^{5}(\vbtwo) + \binom{6}{2}(\vbone)^4(\vbtwo)^2 \nonumber\\
  		&+ \binom{6}{3}(\vbone)^3(\vbtwo)^3 + \binom{6}{4}(\vbone)^2(\vbtwo)^4 \nonumber\\
  		&+ \binom{6}{5}(\vbone)(\vbtwo)^5 + \binom{6}{6}(\vbtwo)^6
  	\end{align}
    Calculate the individual terms in (2). And if you do the calculations correctly, 
    then you should get 
    \begin{align}
    	A &= \vbthree
    \end{align}
  \end{solution}

  \part[2] Do the same for $B$

  \insertQR[-15pt]{}
\begin{solution}[\mcq]
  	If you have done (a), then this shouldn't be too much trouble either. However, do note that 
  	\begin{align}
  		B &= \sum_{k=0}^{6}\binom{6}{k}(\vbone)^{6-k}\cdot\overbrace{(-\vbtwo)^k} \nonumber\\
  		&= \sum_{k=0}^{6}\binom{6}{k}(\vbone)^{6-k}\cdot(\vbtwo)^k\cdot(-1)^k \\
       \Rightarrow B &= \binom{6}{0}(\vbone)^{6} - \binom{6}{1}(\vbone)^{5}(\vbtwo) + 
       \binom{6}{2}(\vbone)^4(\vbtwo)^2 \nonumber\\
  		&- \binom{6}{3}(\vbone)^3(\vbtwo)^3 + \binom{6}{4}(\vbone)^2(\vbtwo)^4 \nonumber\\
  		&- \binom{6}{5}(\vbone)(\vbtwo)^5 + \binom{6}{6}(\vbtwo)^6 \\
  		&= \vbfour
  	\end{align}
  \end{solution}

  \part[1] Calculate $S$ using the expressions found in parts (a) and (b)

  \insertQR[-15pt]{}
\begin{solution}[\mcq]
  	\begin{align}
  		S &= \vbthree + \vbfour = \vbfive
  	\end{align}
  \end{solution}

\end{parts}
