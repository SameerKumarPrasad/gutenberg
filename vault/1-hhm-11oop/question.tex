\question Express $\vbone + \vbtwo + \vbthree$ in the form $\frac{p}{q}$, where $q\neq 0$

\insertQR[-20pt]{}

\watchout

\ifprintanswers
\fi 

\begin{solution}
	\begin{align}
		S = \vbone + \vbtwo + \vbthree &= \vbone \nonumber\\
								& + \vbfour \nonumber\\
								& + \vbfive \\
								&= \vbsix = \vbten \\
		\Rightarrow 10S &=  \vbseven \\
		\text{and } 100S &= \vbeight \\
		\therefore 100S - 10S = \vbnine \Rightarrow S = \WRITEFRAC\vbnine{90}
	\end{align}
	
\end{solution}
