\question Find an expression for the sum of the first $n$ terms of the series 
$\dfrac{3}{4} + \dfrac{5}{36} + \ldots + \dfrac{2n+1}{n^2\cdot(n+1)^2} + \ldots$

\insertQR{}

\ifprintanswers
\fi 

\begin{solution}
	\begin{align}
		S_n &= \sum_{k=1}^{n}\dfrac{2n+1}{n^2\cdot(n+1)^2} = 
		\sum_{k=1}^{n}\dfrac{(n+1)^2 - n^2}{n^2\cdot(n+1)^2} \\
		&= \sum_{k=1}^{n}\dfrac{1}{n^2} - \sum_{k=1}^{n}\dfrac{1}{(n+1)^2} \\
		&= \left( 1 + \dfrac{1}{2^2} + \dfrac{1}{3^2} + \ldots + \dfrac{1}{n^2}\right) 
		- \left( \dfrac{1}{2^2} + \dfrac{1}{3^2} + \ldots + \dfrac{1}{n^2} + \dfrac{1}{(n+1)^2}\right) \\
		&= 1-\dfrac{1}{(n+1)^2} = \dfrac{n\cdot(n+2)}{(n+1)^2}
	\end{align}
\end{solution}
