\SQUARE\vbone\aa
\SQUARE\vbtwo\bb
\SQUARE\vbthree\cc

\ADD\aa\bb\j

\question[3] Find the \textbf{foci},\textbf{vertices} and \textbf{eccentricity} of $\aa x^2 - \bb y^2 = \cc$

\begin{solution}[\halfpage]
  This looks like the equation for a hyperbola. And it can be re-written as
  \begin{align}
    \dfrac{x^2}{\left( \frac{\vbthree}{\vbone}\right)^2}
    - \dfrac{y^2}{\left(\frac{\vbthree}{\vbtwo}\right)^2} &= 1 \\
    \implies \dfrac{x^2}{\underbrace{\vbtwo^2}_{a^2}} - 
    \dfrac{y^2}{\underbrace{\vbone^2}_{b^2}} &= 1
  \end{align}
  
  The vertices of this hyperbola will be on the $x-$axis and the coordinates of the 
  vertices would be $(\pm a,0) = (\pm \vbtwo, 0)$ 
  
  If the coordinates of the foci be $(-c,0)$ and $(c,0)$, then we know that 
  \begin{align}
    c^2 &= a^2 + b^2  =\vbtwo^2 + \vbone^2 =  \j \\
    \implies c &= \sqrt{\j}
  \end{align}
  
  Hence, the foci are at $(\pm\sqrt\j, 0)$. And the eccentricity is
  \begin{align}
    e &= \dfrac{\sqrt{a^2 + b^2}}{a} = \dfrac{c}{a} = \dfrac{\sqrt{\j}}{\vbtwo}  
  \end{align}      
\end{solution}
