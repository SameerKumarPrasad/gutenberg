\begin{parts}
  \part Can the figure shown alongside be its plot? If not, then draw the correct plot

  \begin{solution}
    The given equation - $\vbthree$ - is of the form $\vbfour$, where $p = \vbone$. 
    Hence, the curve cannot look like $A$ but should look like $B$
  \end{solution}

  \part Find coordinates of the focus($F$) and the vertex ($V$). Mark both in the corrected plot 

  \begin{solution}
    The vertex of the parabola is simply at $(0,0)$

    And given the form of the equation - $\vbfour$ - the focus is $\vbfive$
  \end{solution}

  \part Find the length of the latus rectum of the parabola 

  \begin{solution}
    The length of the latus rectum is simply $= \vert 4\times\vbone\vert = \vbsix$
  \end{solution}

\end{parts}
