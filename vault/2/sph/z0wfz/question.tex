\question Differentiate $y=x\sin x$ with respect to $x$ and 
find it's value for $x=\vbone$.

\insertQR{}

\watchout

\begin{solution}
  Simple application of $Product Rule$.
  \begin{align}
    y' &= x \dydx[\sin x] + \sin x\dydx[x] \\
       &= x\cos x + \sin x
  \end{align}
  To find the value of $y'$ for $x=\vbone$ we simply substitute,
  \begin{align}
    y'= \vbtwo
  \end{align}
\end{solution}
