

\question[3] Prove the following identity
\begin{align}
  \dfrac{\cot\theta+\tan\theta}{1+\tan2\theta\tan\theta} 
    = 2\cot2\theta \nonumber
\end{align}

\begin{solution}[\halfpage]
  Expand Left Hand Side (L.H.S.) as follows,
  \begin{align}
    \text{L.H.S} &= \dfrac{\dfrac{1}{\tan\theta}+\tan\theta}
    				  {1+\dfrac{2\tan^2\theta}{1-\tan^2\theta}} \\
    			 &= \dfrac{\dfrac{1+\tan^2\theta}{\tan\theta}}
    			      {\dfrac{1-\tan^2\theta+2\tan^2\theta}
    			      {1-\tan^\theta}} \\
    			 &= \dfrac{1-\tan^2\theta}{\tan\theta} \\
    			 &= \dfrac{2}{\tan2\theta} \\
    			 &= 2\cot2\theta
  \end{align}

\end{solution}
