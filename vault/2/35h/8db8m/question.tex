
\renewcommand\va{60}
\DIVIDE\va{3}\vb
\SUBTRACT\va\vb\vc
\ADD\va\vb\vd

\question[4] Without using a calculator or tables, find the value of the following
\[ \tan\vb^\circ\cdot\tan\vc^\circ\cdot\tan\vd^\circ \]

\begin{solution}[\halfpage]
  If we let $\theta=\vb^\circ$, then note that 
  \begin{align}
    \vc^\circ &= \va^\circ-\theta \\
    \vd^\circ &= \va^\circ+\theta
  \end{align}
  And therefore, the original expression can be re-written as 
  \begin{align}
    &\tan\theta\cdot\tan(\va^\circ-\theta)\cdot\tan(\va^\circ+\theta) \\
    &=\tan\theta\cdot\eTanOfDiff{\va^\circ}{\theta}\cdot\eTanOfSum{\va^\circ}{\theta} \\
    &= \tan\theta\cdot\left[ \dfrac{\tan^2\va^\circ-\tan^2\theta}{1-\tan^2\va^\circ\cdot\tan^2\theta} \right] \\
    &= \tan\theta\cdot\left( \dfrac{3-\tan^2\theta}{1-3\tan^2\theta}\right) 
    = \underbrace{\dfrac{3\tan\theta-\tan^3\theta}{1-3\tan^2\theta}}_{=\tan 3\theta} \\
    &= \tan{60^\circ}=\sqrt{3}
  \end{align}
  
\end{solution}
