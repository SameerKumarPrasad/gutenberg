\ADD\a{2}\b
\SUBTRACT\b{1}\c

\question One and only one of the following four people $John$, $Boris$, 
$Ivan$ or $Stephan$ will win the upcoming $Wimbledon$ tennis championship 
next month. You are told that the chances of $\vbone$ winning are the 
same as $\vbtwo$. $\vbfive$ chances are $\vbseven$ that of $\vbone$. 
$\vbsix$ chances are $\vbeight$ times that of $\vbone$.


\watchout

\begin{parts}
  \part[2] Determine the probability of each of those players individually
  winning the $Wimbledon$ tennis championship.

\begin{solution}[\mcq]
    If the chances of $\vbone$ winning is $x$, then the probabilities 
    for each of them winning are,
    \begin{align}
      P_{\vbone}   &= \dfrac{x}{x+x+2x+4x} = \dfrac{1}{\b} \\
      P_{\vbtwo}   &= \dfrac{x}{x+x+2x+4x} = \dfrac{1}{\b} \\
      P_{\vbthree} &= \dfrac{2x}{x+x+2x+4x} = \dfrac{\vbnine}{\b} \\
      P_{\vbfour}  &= \dfrac{4x}{x+x+2x+4x} = \dfrac{\vbten}{\b}
    \end{align}
  \end{solution}

  \part[1] Find the probability that $\vbone$ will lose the championship.
\begin{solution}[\mcq]
    Probability that $\vbone$ will lose is complement of his probability
    to win. Therefore,
    \begin{align}
      P &= 1 - P_{\vbone} \\
        &= \dfrac{\c}{\b}
    \end{align}    
  \end{solution}

  \part[1] Find the probability of $\vbthree$ or $\vbfour$ winning the 
  championship.
\begin{solution}[\mcq]
    Union of probabilities of $\vbthree$ or $\vbfour$ winning,
    \begin{align}
      P &= P_{\vbthree} + P_{\vbfour} \\
        &= \dfrac{\vbnine}{\b} + \dfrac{\vbten}{\b} \\
        &= \dfrac{\a}{\b}
    \end{align}
  \end{solution}

\end{parts}
