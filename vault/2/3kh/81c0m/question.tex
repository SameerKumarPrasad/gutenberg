
\question Consider the following relational schema,\\
\begin{tabular}{ll}
  Table           &(field1, field2, ...)\\
  \hline
  \texttt{GradStudent}     &(gsid, firstname, lastname, birthdate, address) \\
  \texttt{WorksOn}         &(gsid, pid, hours) \\
  \texttt{ResearchProject} &(pid, pname, lab) \\
  \hline \\
\end{tabular}
where gsid (Grad Student Id) uniquely identifies a Graduate Student and
pid (Project Id) uniquely identifies a Research Project. Now write the 
algebraic expressions for the following queries.

\begin{parts}
  \part Find the first and last names of students who work on 
  projects in the \textit{Graphics} Lab.

  \begin{solution}
    Begin with a select on \texttt{ResearchProject} for \textit{Graphics}
    projects. Project the \textit{pid} from the selection and join with 
    \texttt{WorksOn} to get the \textit{gsid} and finally project from 
    \texttt{GradStudent} to get the first and last names.
    \begin{align}    
      \pi_{firstname, lastname}(\pi_{gsid}(\pi_{pid}(\sigma_{lab="Graphics"}\text{ResearchProject})\bowtie\text{WorksOn})\bowtie\text{GradStudent}) \nonumber
    \end{align}
  \end{solution}

  \part Find the gsid of Graduate Students who work either in
  the \textit{Graphics} lab or in the \text{Databases} lab. 

  \begin{solution}
    Select on \texttt{ResearchProject} and join with \texttt{WorksOn} to get
    \textit{gsid}. Since names are not required, we do not need 
    \texttt{GradStudent}.
    \begin{align}
      \pi_{gsid}((\pi_{pid}\sigma_{lab="Graphics" \vee lab="Databases"}\text{ResearchProject})\bowtie\text{WorksOn}) \nonumber
    \end{align}
  \end{solution}

\end{parts}
