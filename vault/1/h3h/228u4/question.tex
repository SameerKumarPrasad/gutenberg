

\question[2] Find the derivative of $\int_2^{e^x}\dfrac{\ln z}{z}\ud z$ with respect to $x$


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
\fi 

\begin{solution}[\halfpage]
	Let $\ln z = y$. Which means, 
	\begin{align}
		\dfrac{\ud z}{z} &= \ud y \\
		y &= \ln 2 \text{ when } z = 2 \\
		y &= \ln e^x = x \text{ when } z = e^x
	\end{align}
	And therefore,
	\begin{align}
		f(x) = \int_2^{e^x}\dfrac{\ln z}{z}\ud z &= \int_{\ln 2}^{x} y \ud y \\
		&= \left[ \dfrac{y^2}{2}\right]_{\ln 2}^{x} \\
		&= \dfrac{x^2}{2} - \dfrac{(\ln 2)^2}{2} \\
		\Rightarrow \dfrac{\ud}{\ud x}f(x) &= x
	\end{align}
\end{solution}
