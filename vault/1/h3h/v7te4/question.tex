

\question[1] Differentiate $y = (x^2-3x+3)(x^2+2x-1)$


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
\fi 

\begin{solution}[\halfpage]
	\begin{align}
		\dydx &= (x^2-3x+3)\cdot\dfrac{\ud}{\ud x}(x^2+2x-1) + 
		         (x^2+2x-1)\dfrac{\ud}{\ud x}(x^2-3x+3) \\
		      &= (x^2-3x+3)\cdot(2x+2) + (x^2+2x-1)\cdot(2x-3) \\
		      &= 2x^3-6x^2+6x+2x^2-6x+6+2x^3+4x^2-2x-3x^2-6x+3 \\
		      &= 4x^3-3x^2-8x+9
	\end{align}
\end{solution}
