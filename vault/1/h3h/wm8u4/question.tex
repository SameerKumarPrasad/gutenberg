

\question[2] Evaluate $\int x\tan^{-1}x\ud x$


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
  \marginnote[2.25cm]{Integration by parts}
  \marginnote[0.5cm]{Let $y=x^2+1 \Rightarrow x\ud x = \dfrac{\ud y}{2}$}
\fi 

\begin{solution}[\halfpage]
	\begin{align}
		\int x\tan^{-1}x\ud x &= \tan^{-1}x\int x\ud x - \int\dfrac{\ud}{\ud x}\tan^{-1}x\cdot x\ud x \\
		&= \dfrac{x^2}{2}\tan^{-1}x - \int \dfrac{x}{x^2+1}\ud x \\
		&= \dfrac{x^2}{2}\tan^{-1}x - \dfrac{1}{2}\ln(x^2+1) + C 
	\end{align}
\end{solution}
