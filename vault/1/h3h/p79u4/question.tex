

\question[2] Find the derivative of $\int_{x^2}^1 \ln z\ud z$ with respect to $x$


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
\fi 

\begin{solution}[\halfpage]
	Let,
	\begin{align}
		u(z) &= \ln z \\
		\text{and } v(z) &= z \text{ so that } v'(z) = 1 \\
	\end{align}
	And so,
	\begin{align}
		\int\ln z\ud z &= \int u(z)\cdot v'(z)\ud z \\
		&= \underbrace{u(z)v(z) - \int u'(z)v(z)\ud z}_{\text{integration by parts}} \\
		&= z\ln z - \int\dfrac{1}{z}\cdot z\ud z \\
		&= z\cdot(\ln z - 1) + C \\
		\Rightarrow \int_{x^2}^1\ln z \ud z &= \left[ z\cdot(\ln z - 1)\right]_{x^2}^1 \\
		&= x^2 - 2x^2\ln x
	\end{align}
	Which means that the required derivative is 
	\begin{align}
		\dfrac{\ud}{\ud x}(x^2-2x^2\ln x) &= -4x\ln x
	\end{align}
\end{solution}
