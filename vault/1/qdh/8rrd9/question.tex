\question Differentiate $y = \vbone \tan\sqrt{\vbtwo x}$ with respect to $x$

\insertQR{}

\watchout

\ifprintanswers
\fi 

\begin{solution}
  If $z=\sqrt{\vbtwo x}$, then \textbf{chain-rule} tells us that 
  \begin{align}
    \dydx &= \dfrac{\mathrm{d}}{\mathrm{d}z}y\times\dydx[z] \\
    \implies \dydx &= \vbone\sec^2{\sqrt{\vbtwo x}}\cdot\dydx[\sqrt{\vbtwo x}] \\
    &= \vbone\sec^2{\sqrt{\vbtwo x}}\cdot\left( \dfrac{\sqrt\vbtwo}{2} \cdot\dfrac{1}{\sqrt{x}}\right)\\
    &= \dfrac{\vbone\sqrt{\vbtwo}}{2}\left(\dfrac{\sec^2 {\sqrt{\vbtwo x}}}{\sqrt{x}}\right)
  \end{align}
\end{solution}
