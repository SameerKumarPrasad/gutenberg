

\question[3] Find an expression for the sum of the first $n$ terms of the series 
$\dfrac{1}{1\cdot 3} + \dfrac{1}{3\cdot 5} + \dfrac{1}{5\cdot 7} + \ldots$


\ifprintanswers
\fi 

\begin{solution}[\halfpage]
	Each term in the above series is of the form $a_n = \dfrac{1}{(2n-1)\cdot(2n+1)}$. Hence, the
	sum of the first $n$ terms is
	\begin{align}
		S_n &= \sum_{k=1}^{n}\dfrac{1}{(2n-1)\cdot(2n+1)} = 
		\sum_{k=1}^{n}\dfrac{1}{2}\cdot\dfrac{(2n+1)-(2n-1)}{(2n+1)\cdot(2n-1)} \\
		&= \dfrac{1}{2}\cdot\left[ \sum_{k=1}^{n}\dfrac{1}{2n-1} - \sum_{k=1}^{n}\dfrac{1}{2n+1}\right] \\
		&= \dfrac{1}{2}\cdot\left[ \left( 1 + \dfrac{1}{3} + \dfrac{1}{5} + \ldots + \dfrac{1}{2n-1}\right)
		 - \left( \dfrac{1}{3} + \dfrac{1}{5} + \ldots + \dfrac{1}{2n-1} + \dfrac{1}{2n+1} \right)\right] \\
		 &= \dfrac{1}{2}\cdot\left( 1-\dfrac{1}{2n+1}\right) = \dfrac{n}{2n+1}
	\end{align}
\end{solution}
