
\question[3] Find an expression for the sum of the first $n$ terms of the series 
$\dfrac{2}{3} + \dfrac{2}{15} + \dfrac{2}{35} + \ldots + \dfrac{2}{4n^2-1}$


\ifprintanswers
\fi 

\begin{solution}[\halfpage]
	\begin{align}
		S_n &= \sum_{k=1}^{n}\dfrac{2}{4k^2-1} = \sum_{k=1}^{n}\dfrac{(2k+1) - (2k-1)}{(2k+1)\cdot(2k-1)} \\
		&= \sum_{k=1}^{n}\dfrac{1}{2k-1} - \sum_{k=1}^{n}\dfrac{1}{2k+1} \\
		&= \left( 1 + \frac{1}{3} + \frac{1}{5} + \ldots + \frac{1}{2n-1}\right) 
		- \left( \frac{1}{3} + \frac{1}{5} + \ldots + \frac{1}{2n-1} + \frac{1}{2n+1}\right) \\
		&= 1-\dfrac{1}{2n+1} = \dfrac{2n}{2n+1}
	\end{align}
\end{solution}
