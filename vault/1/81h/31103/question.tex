\begin{solution}[\halfpage]
  The required area $A$ - of region $R$ - \asif - is
  \begin{align}
    A &= \int_{\vbfour}^{\vbfive} (\vbone x^2-\vbtwo x+\vbthree)\ud x \\
      &= \left[ \vbone\cdot\dfrac{x^3}{3}-\dfrac{\vbtwo}{2}x^2+\vbthree x\right]_{\vbfour}^{\vbfive} \\
      &= \vbsix
  \end{align}
\end{solution}
