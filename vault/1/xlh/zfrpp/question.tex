\question[3] Find the following limit 
\[ \lim_{n\to\infty}\left( \frac{1^\vbone}{n^\vbtwo} + \frac{2^\vbone}{n^\vbtwo} + \ldots
+ \frac{(n-1)^\vbone}{n^\vbtwo}\right)\]

\watchout[-40pt]

\begin{solution}[\halfpage]
  The above expression can be re-written as 
  \begin{align}
    &\lim_{n\to\infty}\left( \frac{1^\vbone}{n^\vbtwo} + \frac{2^\vbone}{n^\vbtwo} + \ldots
+ \frac{(n-1)^\vbone}{n^\vbtwo}\right) \\ 
  &= \lim_{n\to\infty}\dfrac{1}{n^\vbtwo}\cdot\sum_{k=1}^{n-1}k^\vbone \\
  &= \lim_{n\to\infty}\dfrac{1}{n^\vbtwo}\cdot\vbthree \\
  &= \lim_{n\to\infty}\dfrac{1}{n^\vbtwo}\cdot\vbfour = \vbfive
  \end{align}
\end{solution}
