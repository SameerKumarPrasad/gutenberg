

\question[2] A metal sphere of diameter 42 cm is melted and drawn into a wire of diameter 28 cm.
What would be the length of the drawn wire if 20\% of the metal was lost during the melting process?


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
\fi 

\begin{solution}[\halfpage]
  \begin{align}
    \text{Metal available after melting} &= \text{Volume of drawn wire} \\
    \Rightarrow 0.8\times\dfrac{4}{3}\pi\left(\dfrac{D_{sphere}}{2}\right)^3 &= 
    	\pi\left(\dfrac{D_{wire}}{2}\right)^2\cdot L_{wire} \\
    \Rightarrow L_{wire} &= \dfrac{1.6}{3}\times\dfrac{D_{sphere}^3}{D_{wire}^2} \\
                         &= \dfrac{1.6}{3}\times\dfrac{(42\text{cm})^3}{(28\text{cm})^2} \\
                         &= 50.4\text{ cm}
  \end{align}
\end{solution}
