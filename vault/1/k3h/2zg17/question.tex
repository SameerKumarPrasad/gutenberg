

\question[3] Find the range over which the inequality $\sqrt{(x+14)} < x+2$ holds


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
  \marginnote[2cm]{Only the \textit{positive} roots are considered in this type of question}
\fi 

\begin{solution}[\halfpage]
  If $\sqrt{(x+14)} < x + 2$ then
	\begin{align}
	   x + 14 > 0 &\Rightarrow x > -14 \\
	   x + 2 > 0 &\Rightarrow x > -2 \\
	   \text{Moreover, if } (x + 14) < (x+2)^2 &= x^2+4x+4 \\
	   \text{then } x^2+3x-10 = (x-2)\cdot(x+5) &> 0 \\
	   \Rightarrow x > 2 \text{ or } x < -5 
	\end{align}
	As $x$ has to be $> -2$, $x$ cannot be $< -5$. And therefore, the only feasible solution is 
	$x > 2$
\end{solution}
