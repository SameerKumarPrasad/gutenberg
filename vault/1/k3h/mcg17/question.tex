

\question[1] If $f(x) = 3x^2-4x$, then find the intervals in which it is \textit{increasing} 
and \textit{decreasing}


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
\fi 

\begin{solution}[\mcq]
   $f(x)$ is increasing when $\dfrac{\ud}{\ud x}f(x) > 0$ and decreasing otherwise. Which means,
   \begin{align}
   	   \text{Increasing when }\dfrac{\ud}{\ud x}(3x^2-4x) = (6x-4) > 0 &\Rightarrow x > \frac{2}{3} \\
   	   \text{and decreasing when } (6x-4) < 0 &\Rightarrow x < \frac{2}{3}
   \end{align}
\end{solution}
