

\question[2] If $\tan A = \frac{1}{2}$ and $\tan B = \frac{1}{3}$, then find the value
of $\tan (2A+B)$?


\ifprintanswers
  % stuff to be shown only in the answer key - like explanatory margin figures
\fi 

\begin{solution}[\halfpage]
	\begin{align}
		\tan(2A+B) &= \dfrac{\tan 2A + \tan B}{1-\tan 2A\tan B} \\
		\text{where } \tan 2A &= \dfrac{2\tan A}{1-\tan^2 A} \\
		    &= \dfrac{2\times \frac{1}{2}}{1-\left(\frac{1}{2}\right)^2} = \frac{4}{3} \\
		 \Rightarrow \tan(2A+B) &= \dfrac{\frac{4}{3} + \frac{1}{3}}{1-\frac{4}{3}\cdot\frac{1}{3}} \\
		 &= 3
	\end{align}
\end{solution}
