\ADD\vbone\vbfour\p
\SUBTRACT\vbthree\vbtwo\q
\def\logtwo{0.301}
\EXPR[3]\a{ ((\vbone - (\vbone + \vbfour)*\logtwo) / (\vbtwo + (\vbthree - \vbtwo)*\logtwo)) }

\question Using logarithms, solve the following equation for $x$.
\texttt{Hint:} $\log_{10}5 = \log_{10}\frac{10}{2} = 1 - \log_{10} 2$
\[ 5^{\vbone - \vbtwo x} = 2^{\vbthree x + \vbfour}\]

\watchout[-40pt]
\marginnote[-10pt]{ $\log_{10}2 = 0.301$ }

\begin{solution}
  \begin{align}
    5^{\vbone -\vbtwo x} &= 2^{\vbthree x + \vbfour} \implies 
    (\vbone - \vbtwo x)\cdot\log_{10} 5 = (\vbthree x + \vbfour)\cdot\log_{10} 2 \\
    &\implies (\vbone - \vbtwo x)\cdot(1-\log_{10} 2) = (\vbthree x + \vbfour)\cdot\log_{10} 2 \\
    \text{ or } x &= \dfrac{\vbone - (\vbone + \vbfour)\log_{10} 2}
    {\vbtwo + (\vbthree - \vbtwo)\log_{10} 2} = \a
  \end{align}
\end{solution}
