\DIVIDE\vbone{100}\byhundred
\MULTIPLY\vbtwo\byhundred\a
\MULTIPLY\vbthree\byhundred\b
\MULTIPLY\vbfour\byhundred\c
\MULTIPLY\vbfive\byhundred\ab
\MULTIPLY\vbsix\byhundred\bc
\MULTIPLY\vbseven\byhundred\ca
\MULTIPLY\vbeight\byhundred\abc
\EXPR[0]{\na}{\a-\ab-\ca+\abc}
\EXPR[0]{\aunionbunionc}{\a+\b+\c-\ab-\bc-\ca+\abc}
\SUBTRACT\vbone\aunionbunionc\noabc

\question In a town of $\vbone$ families, \vbtwo\% families buy newspaper $A$, 
\vbthree\% buy newspaper $B$, \vbfour\% buy newspaper $C$, \vbfive\% buy both $A$ and $B$, 
\vbsix\% buy $B$ and $C$ and \vbseven\% buy $A$ and $C$. If \vbeight\% of the families buy all 
three newspapers, then 


\watchout[-50pt]

\begin{parts}
  \part[3] How many families buy newspaper $A$ only? Identify the set in a Venn diagram

  \ifprintanswers
    \begin{marginfigure}
      \begin{venndiagram3sets}
        \fillOnlyA
      \end{venndiagram3sets}
    \end{marginfigure}
  \fi 
\begin{solution}[\halfpage]
    First, let us take stock of what we know 
    \begin{align}
      n(A) = \a, n(B) &= \b, n(C) = \c \\
      n(A\cap B) = \ab, n(B\cap C) &= \bc, n(A\cap C) = \ca \\
      n(A\cap B\cap C) &= \abc
    \end{align}

    What we need are the number of families in the shaded region
    \begin{align}
      n(A_{only}) &= \overbrace{n(A) - n(A\cap B) - n(A\cap C)}^{n(A\cap B\cap C)\text{ subracted twice}} \nonumber \\
      &+ \underbrace{n(A\cap B\cap C)}_{\text{Hence, added once again}} \\
      &= \a-\ab-\ca + \abc = \na
    \end{align}
  \end{solution}

  \part[2] And how many do not buy any? Identify the set in a Venn diagram  

  \ifprintanswers
    \begin{marginfigure}[50pt]
      \begin{venndiagram3sets}
        \fillNotABC
      \end{venndiagram3sets}
    \end{marginfigure}
  \fi 
\begin{solution}[\halfpage]
    The shaded region in the figure alongside represents families that do not buy any newspaper. 
    In set notation form, it is the set $A\cup B\cup C)'$
    \begin{align}
      n(A\cup B\cup C)' &= n(U) - n(A\cup B\cup C)
    \end{align}
    Now, $n(A\cup B\cup C) =  n(A) + n(B) + n(C) - n(A\cap B) - n(B\cap C) - n(C\cap A) + n(A\cap B\cap C)
    = \a + \b + \c - \ab -\bc - \ca + \abc = \aunionbunionc$

    And hence
    \begin{align}
      n(A\cup B\cup C)' &= \vbone - \aunionbunionc = \noabc
    \end{align}
  \end{solution}
\end{parts}
