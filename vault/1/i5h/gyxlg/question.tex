

\question[4] If $\sin x= \dfrac{3}{5}$ and $\cos y = -\dfrac{12}{13}$, then find the values 
of $\sin(x+y)$ and $\sin 2x$ given that both $x$ and $y$ lie in the second quadrant


\ifprintanswers
\fi 

\begin{solution}[\halfpage]
	\begin{align}
		\sin x = \dfrac{3}{5} \text{ when } x \in \left( \frac{\pi}{2}, \pi\right) 
		\Rightarrow \cos x = \sqrt{1 - \sin^2 x} = -\dfrac{4}{5} \\
		\cos y = -\dfrac{12}{13} \text{ when } y \in \left( \frac{\pi}{2}, \pi\right)
		\Rightarrow \sin y = \sqrt{1-\cos^2 y} = \dfrac{5}{13} \\
		\text{And so, }\sin (x+y) = \eSinOfSum{x}{y} \nonumber \\
			= \dfrac{3}{5}\cdot\dfrac{-12}{13} + \dfrac{-4}{5}\cdot\dfrac{5}{13} = \dfrac{-56}{65} \\
	    \text{and } \sin 2x = 2\sin x\cos x = 2\cdot\dfrac{3}{5}\cdot\dfrac{-4}{5} = -\dfrac{24}{25}
	\end{align}
\end{solution}
