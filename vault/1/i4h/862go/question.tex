

\question[3] For what $x$ is $\dfrac{6x-5}{4x+1} < 0$ ?


\ifprintanswers
\fi 

\begin{solution}[\mcq]
	\begin{align}
	   \dfrac{6x-5}{4x+1} &= \dfrac{6}{4}\cdot\underbrace{\left(\dfrac{x-\frac{5}{6}}{x+\frac{1}{4}}\right)}_{\text{focus on this}} < 0 \\
	   \Rightarrow  & \begin{cases}
	   	x > \frac{5}{6} \text{ and } x < \frac{-1}{4} \text{ or, } \\
	   	x > \frac{-1}{4} \text{ and } x < \frac{5}{6}
	   \end{cases}	   
	\end{align}
	Only the second constraint makes sense. And so, $x \in \left(\frac{-1}{4}, \frac{5}{6} \right)$
\end{solution}
