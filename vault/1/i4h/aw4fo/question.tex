\ADD\vbone\vbtwo\a

\FRACADD\vbone{2}\vbtwo{2}\p\d
\FRACMINUS\vbone{2}\vbtwo{2}\b\d
\ABSVALUE\b\q

\FRACADD\vbthree{2}\vbfour{2}\r\d

\ABSVALUE\c\s

\FRACADD\r{2}\q{2}\j\d
\FRACMINUS\r{2}\q{2}\k\d
\ABSVALUE\k\m

\question[3] Prove that $\sin\dfrac{\vbone\theta}{2}\sin\dfrac{\vbtwo\theta}{2} + 
\sin\dfrac{\vbthree\theta}{2}\sin\dfrac{\vbfour\theta}{2} = \sin\j\theta\cdot\sin\m\theta$

\insertQR{QRC}

\ifprintanswers
\fi 

\begin{solution}[\halfpage]
  \begin{fullwidth}
  \begin{align}
      \sin\dfrac{\vbone\theta}{2}\sin\dfrac{\vbtwo\theta}{2} &+ \sin\dfrac{\vbthree\theta}{2}\sin\dfrac{\vbfour\theta}{2}
      = \dfrac{1}{2}\cdot\left[ 2\cdot\sin\dfrac{\vbone\theta}{2}\sin\dfrac{\vbtwo\theta}{2} + 
         2\cdot\sin\dfrac{\vbthree\theta}{2}\sin\dfrac{\vbfour\theta}{2} \right] \\
       &= \dfrac{1}{2}\left[ \left\lbrace \cos\left( \dfrac{\vbone\theta}{2} + \dfrac{\vbtwo\theta}{2}\right) 
           - \cos\left( \dfrac{\vbtwo\theta}{2} - \dfrac{\vbone\theta}{2}\right)\right\rbrace
           + \left\lbrace \cos\left( \dfrac{\vbthree\theta}{2} + \dfrac{\vbfour\theta}{2}\right) 
           - \cos\left( \dfrac{\vbthree\theta}{2} - \dfrac{\vbfour\theta}{2}\right)\right\rbrace
           \right] \\
       &= \dfrac{1}{2}\left[ (\cos \p\theta - \cos\q\theta) + ( \cos\r\theta - \cos\s\theta)\right] = 
       \dfrac{1}{2}\left( \cos\r\theta - \cos\q\theta \right) \\
       &= \dfrac{1}{2}\cdot 2\cdot\sin\left( \dfrac{\r\theta + \q\theta}{2}\right)\sin\left( \dfrac{\r\theta - \q\theta}{2}\right) \\
       &= \sin\j\theta\cdot\sin\m\theta
  \end{align}
  \end{fullwidth}
\end{solution}
