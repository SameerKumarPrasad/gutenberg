

\question[2] The sum of an infinite geometric progression is 243 and the sum of
its first five terms is 275. Find the progression


\ifprintanswers
\fi 

\begin{solution}[\halfpage]
	\begin{align}
		\dfrac{a}{1-r} &= 243 \\
		\eSumOfGP{r}{5} &= 275 \Rightarrow 243\cdot(1-r^5) = 275 \\
		\Rightarrow r^5 &= 1 - \dfrac{275}{243} = -\dfrac{32}{243} \\
		\therefore r &= -\dfrac{2}{3} \\
		\text{ and } \dfrac{a}{1-\left(-\frac{2}{3}\right)} &= 243 \Rightarrow a = \dfrac{5\times 243}{3} = 405
	\end{align}
\end{solution}
